class Node
{
    public:
        int data;
        Node *next;
};

Node *head = new Node();

void insert(int new_data)
{
    Node *new_node = new Node();
    new_node->data = new_data;
    new_node->next = head;
    head = new_node;
}
void display()
{
    Node *traverse = new Node();
    traverse = head;
    while(traverse!=NULL)
    {
        cout<<traverse->data<<" ";
        traverse = traverse->next;
    }
}
